رحـاب و احـباب
فتاتين تتهامس دون جـواب
يكون ده اعجـاب ولا عتاب
سالت قلبــى قلبـي داب
دخ فــيه روح الشــباب
الهوى بالســـهام والنشاب
بس بالنظـرات دون اسباب
والله حكـمة تشـقل الالباب
****
مرحبا يا رحاب ويا احباب
نعبر الحاجز والصــعاب
نفـتح للامــــل ابواب
نترك الوهـــم والسراب
نبـدا الريد دون حــجاب
نرحل لمــدن الضـباب
****
فـؤادى مــنازل احباب
عايش فـى اسىْ وعـزاب
من الجرح القـديم ماطاب
بجمالكن زرعتاً تيــراب
سجد راكـــــعاً واناب
ورقم الـــلهفة كان تاب
