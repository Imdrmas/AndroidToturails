لو بيدي كنت طاوعت الليالى
واجــــــبرت الايام تحـــلالى
من الوهن غيرت احـــــوالى
اصطفيت الحبيب الغـــــــالى
عشت الهـــــوى المـــــــثالى
وسكنت العـــــلا والعــــلالى
[٢]
لوبيدي اشيل دفاتر احزانــى
المليانه شجــــــن ومعــــانى
اعـــزفة الحــــان واغانــــى
لحبيبة عمــــري الرجيانـــي
ومــــا تركت الالم افــــنانى
ولا الهجــــــــــر اضـــنانى
[٣]
لو بيدي احمل هموم غربتى
ارحــــــل للبلد دون دنــيتى
عشـــــان تـــــزداد قـسوتـى
ابكى وحـــدي تلين حــرقتى
واحــــــكى للتاريخ قصـــتى
[٤]
لوبيدي ابوح بصمتى الرهيب
فـــــى ليل الشجـــن الغــريب
اسكب البـــــــكاء والنحــــيب
دمــوع غـــزيرة فاقدة حـبيب
تحلــم بالبلــــسم والطـــــبيب
[٥]
لو بيدي اجيك بكل الجــراح
طالب الرضـــاء والصــراح
كايس الهـــناء والانشـــراح
القــــى المـــــنى وارتــــاح
اغـــنى ليك كلبلـــبل صداح
[٦]
لوبيدي اتقصص بالايام الجامدة
افـتـش دواخلـــــــى الهــــامدة
اتعمق فـى مكامــينى الصـامدة
اجج نيــران شوقي الصـــاعدة
[٧]
لوبيدي انثر هلــع الاوجـاع
ابحثرها فنــــــون وابـــــداع
وافتح نوافيزي للاخـــــتراع
تتفرهد احاسيسى من النزاع
[٨]
لو بيدي الجأ بنفسى الشاردة
لعوالـــم الحــــــــب الباردة
احكى الحكــــــاوى الساردة
اوقف نزيف جفونى الواردة
اداوى كـــــــــــبدى الهاردة
[٩]
لوبيدي سانتحب ملء اعماقى
اريــح ضمـيري واشــــواقى
وابعد مــن دورة الســـــواقى
[١٠]
لوبيدي تغترب بوجهك منى اكثر
تغمرني بنور وجنتيك الاخضــر
والمس جمــــــــــــــالك الانضر
نفرح بحلاوة عــــــــــودة السفر
نتعـــــــانق حــــــــــــد الفــــتر
أتذوق قبـــــــــلات الســــــــمر
اتوه فــــــــى شــــــــــهد الشعر
عاشـــــق يتحــــــدى الســــــهر
واقول حروف ما قالها قبلي بشر
