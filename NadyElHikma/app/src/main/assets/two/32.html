"نغتني بفضل جهودنا ويتزايد غنانا بفضل اقتصادنا."

"ينظر الإنسان في وجه أخيه الإنسان، وينظر الله في قلبه."

"من لم يمتحن الألم يجهل ثمن السعادة."

"من يتكل على الأمل يمت جوعا."

"العسل شيء، وسعر العسل شيء آخر."

"لا تبع سمكا مازال في البحر."

"من يتردد بين مسجدين يرجع دون صلاة."

"كن سيد من لا تحبه وعبد من تحبه."

"العاشق هو من لا يترك سيره أثرا على الثلج."

"حب وردة جعل البستاني ألف شوكة."

"من لديه أفكار مهوسة لن يبلغ المساء."

"من يوسع خطواته لن يصل إلى الباب."

"من يبحث عن السلام عليه ان يكون أطرش وأعمى وأخرس"

"السلاح عو صاحبه."

"ألف رجل مسلح لا يستطيعون تجريد رجل عار."

"نهرب من المطر، فنصادف البرد."

"يعطي الله الخبز أحدهم والشهية لآخر."

"كي تحك ينبغي أن تكون لك أظافر."

"قبطانان على ظهر سفينة، أسهل وسيلة لإغراقها."

"عندما تمطر السماء ينبغي إملاء الجرار."

"نصف طبيب يفقدك صحتك، نصف إمام يفقدك إيمانك."

"الكلب الذي يريد العض لا يبرز أسنانه."

"للأرض آذان وللريح صوت."

"تعلم التضحية بلحيتك كي تنقذ رأسك."

"من مسك لجامه يعرف الحصان خياله."

"يذهب الدين وأنت ضاحك، ويعود إليك وأنت باك."

"من يحبل في السر يلد في العلانية."

"الإنسان الأسعد: من المهد إلى اللحد."

"الصبر مفتاح الجنة."

"من يبحث عن صديق لا عيب فيه يبق دون صديق."

"يمكننا التخلص من صديق بكلمة واحدة، لكن ألف كلمة لا تكفي لاكتسابه."

"إذا تآمر العالم بأجمعه عليك، لا يستطيع الإساءة إليك أكثر من ربع ما تسيئه غلى نفسك."

"اليوم «صيام» قال الهر عندما لم يستطع الحصول على شحك الخنزير."

"لكل زهرة عبيرها."

"الطموح مرض لا دواء له إلا حفنة من التراب."

"حيث يفتقد الخروف تدعى العنزة صاحبة الجلالة."

"غرابي ولا عندليب الغير."

"يعرف الحصان من خياله، والمرأة من رجلها."

"ساعة عدل تساوي سبعين عاما من الصلاة."

"العدالة نصف الدين."

"عندما يدخل الثأر من الباب، تخرج العدالة من المدخنة."

"كن قاطع طرق، كن لصا، لكن ابق عادلا."

"من يطلب عطاء يحمر مرة واحدة، ومن يرفض العطاء يحمر مرتين."

"كل العنب ولا تطلب الكرمة."

"أن تعطي الغني كأنك تحمل الماء إلى البحر."

"كل الأيام أعياد بالنسبة إلى المجانين."

"لا يأتي العقل مع العمر، بل هو في الرأس."

"من يبك من أجل الجميع ينته بفقدان بصره."

"ابتعد عن غضب الرجل الهادئ وعن انقضاض الحيوان المروض."

"الفقر قميص من نار."

"عندما يقع الغني يقال: هذا حادث، وعندما يقع الفقير يقال: إنه سكران."

"عندما يتم القدر، تغشى عين الحكمة."

"القلب طفل يتمنى ما يشتهيه."

"لحم الإنسان لا يؤكل، وجلده لا يلبس، فماذا فيه عدا ذلك غير حلاوة اللسان."

"من كبح لسانه حفظ رأسه."

"يرتبط الإنسان بكلامه، كما يرتبط الحيوان بقيده."

"ليس للسان عظام لكنه يكسر العظام."

"الكلمة السيئة والعملة المزيفة تعودان إلى صاحبهما."

"الكلب الجائع لا يهاب الأسد."

"من يأكل وبطنه ملآن يحفر قبره بأسنانه."

"صحة بدون مال مرض بدون وجع."

"امدح النهار عندما يطل المساء."

"امدح الإنسان بعد موته."

"المرأة التي تأتي معها بعرش مذهب تجلس هي عليه."

"نصيحة المرأة لا تنفع سوى المرأة."

"عندما يطفأ النور جميع النساء تتشابه."

"شاهد أولا، ثم اعرف."

"من فرط وقوعنا في الخطأ نتعلم."

"من يعرف كثيرا يخطئ كثيرا."

"لا يقول العاقل ما يعرف، ويقول المغفل ما لا يعرف."

"ماذا ينفعني وسع العالم إذا كان حذائي ضيقا."

"يموت الحصان ويبقى سرجه، ينتهي الإنسان ويبقى اسمه."

"شيئان لا تستطيع التحديق بهما: الشمس والموت."

"الموت جمل أسود يركع أمام جميع البواب."

"من قبل النصيحة اجتاز الجبل، ومن لم يقبلها أخطأ طريقه حتى في السهل."

"لكل ليل نهار، ولكل شتاء صيف."

"ما لم يحصل في عام، قد يحصل في أحد الأيام."

"يمضي النهار وتمر الحياة، والغبي يفرح باقتراب عيد مولده."

"عندما يفقد العسل يكتفي بالدبس."

"يغتني المرء بالتوفير."

"إذا فسدت السياسة ذهب السلطان"

"إذا كان السارق يكذب، فغن السرقة لا تكذب."
