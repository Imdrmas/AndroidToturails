حكم عن الصبر

"يانفسُ صبراً فعُقْبَى الصبرِ صالحةٌ … لا بدَّ أن يأتي الرحمنُ بالفرجِ"

مثل عربي

"كم بين صبرٍ غدا للذلِّ مُجْتلباً … وبينَ صبْرٍ غدا للعزِّ يجتلبُ"

إِبراهيم اليازجي

شاعر

إِذا عِلَ صبرُ المرءِ فيما ينوبُهُ … فلا بدَّ من أن يستكينَ ويَجْزعا

ابن الدهان الموصلي

"ما أشقى من لا صبر لهم"

شكسبير

"إِذا لم تستطعْ للرزءِ دَفْعاً … فصبراً للرزيةِ واحتسابا فما نالَ المنى في العيشِ إِلا … غبيَّ القوم أو فَطِنٌ تغابى"

الشريف المرتضى

"يا نفسُ صَبْراً على ما قد منيتِ به … فالحرُّ يصبرُ عند الحادثِ الجَلَلِ"

الشاغوري

"فصبراً فليسَ الأجرُ إِلا لصابرٍ … على الدهرِ إِن الدهرَ لم يخلُ من خَطْبِ"

ابن حميدس

"الصبرُ أولى بوقارِ الفتى … من قلقٍ يهتكُ سترَ الوقار"

غانم المالقي

"إذا وجدت الصبر يساوي البلادة في بعض الناس فلا تخلطن بين تبلد الطباع المريضة وبين تسليم الأقوياء لما نزل بهم"

محمد الغزالي

"اشد ساعات اليوم ظلمةً هي تلك التي تسبق طلوع الشمس"

ابن سينا

"“حين يشتد الوجع، ويتصاعد الألم، ليس هناك علاج فوري وفاعل مثل وصفة الصبر والصلاة لتهدأ النفس وتعود إلى طبيعتها”"

جلال الخوالدة

"لا بد من الصبر ليجتاز العابدون البلاء"

شكسبير

"صبر قليل يساوى راحة عشر سنين"

مثل انجليزي

محمد الغزالي

"الصبر هو الفضيلة الأولى التي يتعلّمها الراعي"

مثل فارسي

"العجلة الوديعة ترضع بحلمتين."

"قطرة بعد قطرة يمتلئ البرميل"

مثل قتلوني

"السير على مهل يوصل إلى البعيد"

مثل مجري

"عجبا للمؤمن لا يقضي الله له قضاء إلا كان خيرا له، إن أصابته ضراء صبر فكان خيرا له، وإن أصابته سراء شكر فكان خيرا له"

محمد رسول الله صلى الله عليه و سلم
